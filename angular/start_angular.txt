Install -> 

	To install npm ->
		Install nodejs 
			https://nodejs.org/en/download/
		and npm will be installed by-default
		
		$ node -v
		$ npm -v
		
	Install angular/cli package
		$ npm install -g @angular/cli
		$ npm i -g @angular/cli
		$ npm install @angular/cli -g
		$ npm i @angular/cli -g
			latest angular (13.x) requires latest node (16+)
		
		$ npm i -g @angular/cli@x.y
		
		$ ng version
		
	Create angular project
		$ ng new angular-tutorial
